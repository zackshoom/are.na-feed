<script>

// Are.na channel slug

const channel = "zackshoom-com";


// Fetch Are.na blocks

async function loadArenaBlocks() {

  try {

    const res = await fetch(`https://api.are.na/v2/channels/${channel}?per=100&_=${Date.now()}`);

    const data = await res.json();

    const grid = document.getElementById("arena-grid");

    grid.innerHTML = ""; // Clear old blocks


    // Newest first

    const blocks = data.contents.slice().reverse();


    blocks.forEach(block => {

      let item = document.createElement("div");

      item.className = "arena-item";


      // Image block

      if(block.class === "Image" && block.image?.display?.url){

        const url = `${block.image.display.url}?_=${Date.now()}`;

        item.innerHTML = `<img src="${url}" />`;

        item.onclick = () => {

          const modal = document.getElementById("arena-modal");

          const modalImg = document.getElementById("arena-modal-img");

          modalImg.src = url;

          modal.style.display = "flex";

        }

      }

      // Link block

      else if(block.class === "Link" && block.source?.url){

        const url = block.source.url;

        item.innerHTML = `

          <a href="${url}" target="_blank">

            <img src="https://api.microlink.io/?url=${url}&screenshot=true&meta=false&embed=screenshot.url&_=${Date.now()}">

          </a>

        `;

      }

      // Text block

      else if(block.class === "Text" && block.content){

        item.innerHTML = `<div class="arena-text">${block.content}</div>`;

      } else {

        return; // skip unusable blocks

      }


      grid.appendChild(item);

    });


  } catch(err) {

    console.error("Are.na fetch error:", err);

  }

}


// Initial load

loadArenaBlocks();


// Optional: auto-refresh every 60 seconds

setInterval(loadArenaBlocks, 60000);

</script>
